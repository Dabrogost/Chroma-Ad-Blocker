// Run in the extension service worker console.
(async () => {
  const {
    config,
    statsV2,
    subscriptions = [],
    appliedNetworkRuleCount = 0,
    appliedNetworkRulesPerSub = {},
    subscriptionCosmeticRules = [],
    subscriptionScriptletRules = []
  } = await chrome.storage.local.get(null);

  const manifestRulesets = chrome.runtime.getManifest().declarative_net_request.rule_resources;
  const dynamicRules = await chrome.declarativeNetRequest.getDynamicRules();
  const enabledRulesets = await chrome.declarativeNetRequest.getEnabledRulesets();

  const defaultDynamicRules = dynamicRules.filter(rule => rule.id >= 1000 && rule.id <= 99999);
  const subscriptionRules = dynamicRules.filter(rule => rule.id >= 100000 && rule.id < 9000000);
  const whitelistRules = dynamicRules.filter(rule => rule.id >= 9000000);

  console.group('Chroma Ad-Blocker Rule Audit');

  console.log('1. Static DNR rulesets');
  console.table(manifestRulesets.map(resource => ({
    id: resource.id,
    path: resource.path,
    manifestEnabled: resource.enabled !== false,
    currentlyEnabled: enabledRulesets.includes(resource.id)
  })));

  console.log('2. Dynamic DNR rules');
  console.table({
    defaultDynamicRules: defaultDynamicRules.length,
    subscriptionRules: subscriptionRules.length,
    whitelistRules: whitelistRules.length,
    totalDynamicRules: dynamicRules.length,
    appliedNetworkRuleCount
  });

  console.log('3. Subscription health');
  console.table(subscriptions.map(subscription => ({
    name: subscription.name,
    enabled: subscription.enabled,
    network: subscription.ruleCount?.network || 0,
    cosmetic: subscription.ruleCount?.cosmetic || 0,
    scriptlet: subscription.ruleCount?.scriptlet || 0,
    appliedNetwork: appliedNetworkRulesPerSub[subscription.id] || 0,
    lastUpdate: subscription.lastUpdated ? new Date(subscription.lastUpdated).toLocaleString() : 'never',
    lastError: subscription.lastError || ''
  })));

  console.log('4. Cosmetic and scriptlet layers');
  console.table({
    subscriptionCosmeticRules: subscriptionCosmeticRules.length,
    subscriptionScriptletRules: subscriptionScriptletRules.length,
    uniqueScriptlets: new Set(subscriptionScriptletRules.map(rule => rule.scriptlet)).size
  });

  console.log('5. Stored state');
  console.table({
    enabled: config?.enabled,
    networkBlocking: config?.networkBlocking,
    protectionEvents: statsV2?.totals?.protectionEvents || 0,
    networkBlocks: statsV2?.totals?.networkBlocks || 0
  });

  console.groupEnd();
})();
